import pandas as pd 

datos = [
    {"id": 1, "nombre": "Ana", "edad": 20, "nota": 4.5},
    {"id": 2, "nombre": "Luis", "edad": 22, "nota": 3.8},
    {"id": 3, "nombre": "Pedro", "edad": 19, "nota": 4.9}

df = pd.DataFrame(datos)

filtro = df[df["nota"] > 4.0]
print(filtro)
